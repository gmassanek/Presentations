jQuery Events

//Old (and Current) Way

var handler = function(){ ... };

$('selector').bind('click', handler);

$('selector').click(handler);

$('selector').live('click', handler);

$('selector').delegate('sub-selector',
                       'click',
                       handler);

jQuery Events

//New Way

var handler = function(){ ... };

$('selector').on('click', handler);

$('selector').on('click',
                 'sub-selector',
                 handler);

$('selector').off('click', handler);

jQuery Selectors / CSS3

//Simple Selectors
$('*');
$('a');
$('#id');
$('.class');
$('.class.class2');

//Compound Selectors
$('#id .class');
$('#id > .class');
$('#id, #otherid');

//Psuedo Selectors
$('tr:even');
$('input:enabled');
$('div:visible');
$('li:last-child');
$('div:not(.class)');

//Attribute Selectors
$('div[id*="contains"]');
$('div[id$="endswith"');
$('div[id^="startswith"');
